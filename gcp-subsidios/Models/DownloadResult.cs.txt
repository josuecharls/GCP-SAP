namespace gcp_subsidios.Models
{
    internal class DownloadResult
    {
        public bool Success { get; init; }

        public string? LocalFilePath { get; init; }

        public ulong FileSizeBytes { get; init; }

        public string? ErrorMessage { get; init; }

        public static DownloadResult CreateSuccess(string localFilePath, ulong fileSizeBytes) =>
            new() { Success = true, LocalFilePath = localFilePath, FileSizeBytes = fileSizeBytes };

        public static DownloadResult CreateFailure(string errorMessage) =>
            new() { Success = false, ErrorMessage = errorMessage };
    }
}
